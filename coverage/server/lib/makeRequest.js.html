function getMeetings(args) {

  var API_KEY = args.iparams.API_KEY;

  //Payload for jwt.
  var payload = {
    iss: API_KEY,
    exp: new Date().getTime() + 60,
  };
  return {
    method: "GET",
    url: "https://api.zoom.us/v2/users/" + HOST_ID + "/meetings",
    body: JSON.stringify({ topic: args.topic }),
    headers: {
      Authorization: "Bearer " + jwt.sign(payload, API_SECRET),
